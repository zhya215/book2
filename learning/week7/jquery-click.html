})

$('button#twobars').click(function(){
    var svg = "<svg><rect height='50' width='10'/><rect height='50' width='10' x='20'/></svg>"
    $('.myviz').html(svg)
})


$('button#fivebars').click(function(){
    var svg = "<svg>"
    for(i=0; i < 5; i++){
        svg+="<rect height='50' width='10' x='"+(i*20)+"'/>"
    }
    svg+="</svg>"
    $('.myviz').html(svg)
})
